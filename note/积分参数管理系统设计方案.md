## 一、问题清单

| # | 问题描述 | 根因 | 状态 |

### 2.2 关键规则

1. **prop_list 顺序 = point_list key 顺序**
   - key 格式：`duration_id, audio_id, ratio_id, resolution_id`

2. **prop_value_id 在同一模型内唯一，不同模型间独立**
   - 模型 A 的 480p → id = 267
   - 模型 B 的 480p → id = 293

3. **prop_value_name 通常保持一致（跨模型迁移锚点）**
   - 不同模型的相同选项 name 相同（如 "480p", "1:1"）

## 三、问题根因分析

### P1: 积分 key 排序错误

**当前错误代码**：
```javascript
// ❌ 按数字大小排序 - 错误！
const sortedValueIds = [...valueIds].sort((a, b) => Number(a) - Number(b));
const combineKey = sortedValueIds.join(","); // "246,258,267,260" ❌
```

**正确做法**：
```javascript
// ✅ 按 prop_list 下标顺序拼接 - 正确！
const ids = propList.map(prop => paramValues[prop.prop_str]);
const combineKey = ids.join(","); // "246,258,260,267" ✅
```

### P2: 切换模型后积分错误

**当前问题**：切换模型只更新 `model_id`，不迁移参数，导致旧 id 残留。

**正确做法**：切换模型时，智能迁移参数。

### P3: 不存在的选项仍显示

**当前问题**：选项直接从 `prop_values_list` 渲染，未过滤。

**正确做法**：基于当前模型的 prop_list 过滤选项。

## 五、工具函数实现

### 5.1 modelAdapter.js

```javascript
/**
 * 积分参数管理工具
 * 统一处理后端数据适配、积分计算、参数迁移
 */

import { getParamValueChinese } from "./paramsMap.js";

/**
 * 推断参数类型
 * @param {number} viewType - 后端 prop_viewtype
 * @param {string} propStr - 参数键名
 * @returns {'duration'|'ratio'|'resolution'|'switch'|'radio'}
 */
function inferViewType(viewType, propStr) {
  if (propStr === "duration") return "duration";
  if (propStr === "aspect_ratio") return "ratio";
  if (propStr === "resolution_tier") return "resolution";
  if (viewType === 4) return "switch";
  return "radio";
}

/**
 * 适配后端模型数据
 * @param {Object} raw - 后端原始模型数据
 * @returns {Object} 适配后的模型规范
 */
export function adaptModel(raw) {
  if (!raw) return null;

  const propSpecs = raw.prop_list.map((prop, index) => ({
    propId: prop.prop_id,
    propKey: prop.prop_str,
    propName: prop.prop_name,
    viewType: inferViewType(prop.prop_viewtype, prop.prop_str),
    options: prop.prop_values_list.map((v) => ({
      id: v.prop_value_id,
      name: v.prop_value_name,
      label: getParamValueChinese(v.prop_value_name),
    })),
    keyIndex: index,
  }));

  return {
    id: raw.id,
    name: raw.model_name,
    type: raw.model_type,
    company: raw.model_company,
    frame: raw.model_frame,
    propSpecs,
    pointMap: raw.point_list || {},
  };
}

/**
 * 构建积分 key（按 propSpecs 下标顺序）
 * @param {Object} spec - 适配后的模型规范
 * @param {Object} values - 当前参数值
 * @returns {string} 格式：id1,id2,id3,...
 */
export function buildPointKey(spec, values) {
  if (!spec) return "";

  return spec.propSpecs
    .map((p) => {
      const val = values[p.propKey];
      return val == null || val === "" ? "" : String(val);
    })
    .join(",");
}

/**
 * 计算积分
 * @param {Object} spec - 适配后的模型规范
 * @param {Object} values - 当前参数值
 * @returns {{ point: number, key: string, hit: boolean }}
 */
export function calcPoint(spec, values) {
  if (!spec) return { point: 0, key: "", hit: false };

  const key = buildPointKey(spec, values);
  const point = spec.pointMap[key];

  if (point !== undefined) {
    return { point, key, hit: true };
  }

  return { point: 0, key, hit: false };
}

/**
 * 切换模型时迁移参数
 * @param {Object|null} oldSpec - 旧模型规范
 * @param {Object} oldValues - 旧参数值
 * @param {Object} newSpec - 新模型规范
 * @returns {Object} 新参数值
 */
export function migrateParams(oldSpec, oldValues, newSpec) {
  const result = {};

  for (const newProp of newSpec.propSpecs) {
    const newPropKey = newProp.propKey;
    const oldValue = oldValues?.[newPropKey];

    // 情况1：旧值存在且在新模型中有效 → 直接复用
    if (oldValue && newProp.options.some((o) => o.id === oldValue)) {
      result[newPropKey] = oldValue;
      continue;
    }

    // 情况2：旧值存在但在新模型中无效 → 按 name 匹配
    if (oldValue) {
      let oldName = null;

      if (oldSpec) {
        const oldProp = oldSpec.propSpecs.find(
          (p) => p.propKey === newPropKey
        );
        if (oldProp) {
          const oldOption = oldProp.options.find((o) => o.id === oldValue);
          if (oldOption) oldName = oldOption.name;
        }
      }

      if (oldName) {
        const matchedOption = newProp.options.find(
          (o) => o.name === oldName
        );
        if (matchedOption) {
          result[newPropKey] = matchedOption.id;
          continue;
        }
      }
    }

    // 情况3：找不到匹配 → 使用新模型默认值
    result[newPropKey] = newProp.options[0]?.id;
  }

  return result;
}

/**
 * 获取参数的有效选项列表
 * @param {Object} spec - 模型规范
 * @param {string} propKey - 参数键名
 * @returns {Array} 有效选项列表
 */
export function getValidOptions(spec, propKey) {
  if (!spec) return [];
  const paramSpec = spec.propSpecs.find((p) => p.propKey === propKey);
  return paramSpec?.options || [];
}

/**
 * 获取参数的中文标签（用于 summary 展示）
 * @param {Object} spec - 模型规范
 * @param {Object} values - 当前参数值
 * @param {string} propKey - 参数键名
 * @returns {string|null} 中文标签或 null
 */
export function getParamLabel(spec, values, propKey) {
  if (!spec || !values) return null;

  const val = values[propKey];
  if (val == null || val === "") return null;

  const paramSpec = spec.propSpecs.find((p) => p.propKey === propKey);
  if (!paramSpec) return null;

  const option = paramSpec.options.find((o) => o.id === val);
  return option?.label || null;
}
```

## 八、注意事项

### 8.1 时序问题

模型数据是异步加载的，确保在 `modelList.length > 0` 后再进行操作：

```javascript
useEffect(() => {
  if (modelList.length > 0) {
    initNodeDefaults();
  }
}, [modelList.length]);
```

### 8.2 积分未命中处理

当 `calcPoint` 返回 `hit: false` 时，表示当前参数组合在 `point_list` 中不存在：
- UI 可显示警告样式
- 提交时可禁用按钮并提示用户

### 8.3 必填参数检查

可通过 `spec.propSpecs[].required` 标记必填参数，提交前检查：

```javascript
const hasRequiredParams = spec?.propSpecs.every((p) => {
  if (!p.required) return true;
  const val = paramValues[p.propKey];
  return val != null && val !== "";
});
```
